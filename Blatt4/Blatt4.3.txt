within the object. These functions have to be called as follows:
companies.addNew("Adobe", "ADBE");

		const companies = {
				data: [
				{ name: "Google", id: "GOOGL" },
				{ name: "Microsoft", id: "MSFT" },
				{ name: "Oracle", id: "ORCL" }
				],
				count: 3
		};

		companies.addNew = function(name, id) {
			this.data.push({"name": name, "id": id});
			this.count++;
		}
		companies.delete = function(id){
			for(let i = 0; i < this.count;i++){
				if(this.data[i].id === id){
					this.data.splice(i,1);
					this.count--;
				}
			}
		}
